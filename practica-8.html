⚖️ 12. Diferencias clave (explicación profunda)

  • Centralización del código en main.py, con mayor control del flujo y menor fragmentación.

🔥 13. ¿Qué es Firebase?

Firebase es una plataforma que permite autenticación, base de datos y backend sin servidor.

🗄️ 14. ¿Qué es Firestore?

Es una base de datos NoSQL en la nube que guarda datos en documentos y se consulta mediante API.

🔐 15. ¿Qué es autenticación anónima?

Permite crear sesión sin usuario real y obtener un identificador único (UID).

🌐 16. ¿Qué es una REST API?

Es una forma de comunicación entre sistemas usando HTTP.

  • GET → obtener datos.
  • POST → enviar datos.
  • PATCH → actualizar.
  • DELETE → eliminar.

🧪 17. Configuración de Firebase (paso a paso)

  • Crear proyecto en Firebase Console.
  • Registrar Web App desde el ícono web y asignar nombre.
  • Obtener configuración para FIREBASE_API_KEY y FIREBASE_PROJECT_ID.
  • Activar autenticación anónima en Authentication → Anonymous → Enable.
  • Crear Firestore en modo prueba.
  • Configurar reglas para que cada usuario solo pueda leer y escribir sus propios favoritos usando su uid.
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{userId}/favorites/{docId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}

def fetch_json(url: str, headers: dict | None = None) -> dict:
   request_headers = {"User-Agent": "PokeDesk-Flet/1.0"}
   if headers:
       request_headers.update(headers)


   request = urllib.request.Request(url, headers=request_headers)
   try:
       with urllib.request.urlopen(request, timeout=20) as response:
           return json.load(response)
   except urllib.error.HTTPError as exc:
       body = exc.read().decode("utf-8", errors="ignore")
       raise RuntimeError(f"HTTP {exc.code} al consultar {url}: {body}") from exc
   except urllib.error.URLError as exc:
       raise RuntimeError(f"No se pudo cargar información desde {url}") from exc
   except Exception as exc:
       raise RuntimeError(f"Error inesperado al cargar {url}: {exc}") from exc




def post_json(url: str, payload: dict, headers: dict | None = None) -> dict:
   data = json.dumps(payload).encode("utf-8")
   request_headers = {
       "Content-Type": "application/json",
       "User-Agent": "PokeDesk-Flet/1.0",
   }
   if headers:
       request_headers.update(headers)


   request = urllib.request.Request(
       url,
       data=data,
       headers=request_headers,
       method="POST",
   )
   try:
       with urllib.request.urlopen(request, timeout=20) as response:
           return json.load(response)
   except urllib.error.HTTPError as exc:
       body = exc.read().decode("utf-8", errors="ignore")
       raise RuntimeError(f"HTTP {exc.code}: {body}") from exc
   except Exception as exc:
       raise RuntimeError(f"Error en POST {url}: {exc}") from exc




def patch_json(url: str, payload: dict, headers: dict | None = None) -> dict:
   data = json.dumps(payload).encode("utf-8")

def format_pokemon_name(name: str) -> str:
   return name.replace("-", " ").title()




def translate_type(type_name: str) -> str:
   return TYPE_LABELS.get(type_name, type_name)




def translate_stat(stat_name: str) -> str:
   return STAT_LABELS.get(stat_name, stat_name.replace("-", " "))




async def fetch_pokemon_details(url: str) -> dict:
   return await asyncio.to_thread(fetch_json, url)




async def build_pokemon_collection_async() -> list[Pokemon]:
   payload = await asyncio.to_thread(fetch_json, API_URL)
   urls = [item["url"] for item in payload.get("results", [])]
   semaphore = asyncio.Semaphore(6)


   async def fetch_limited(url: str) -> dict:
       async with semaphore:
           return await fetch_pokemon_details(url)


   details_list = await asyncio.gather(
       *(fetch_limited(url) for url in urls),
       return_exceptions=True,
   )


   collection: list[Pokemon] = []


   for details in details_list:
       if isinstance(details, Exception):
           continue


       display_name = format_pokemon_name(details["name"])
       artwork = details["sprites"]["other"]["official-artwork"]["front_default"]
       sprite = details["sprites"]["front_default"]


       collection.append(
           Pokemon(
               id=details["id"],
               name=details["name"],
               display_name=display_name,
               image=artwork or sprite or "",
               height=details["height"] / 10,
               weight=details["weight"] / 10,
               base_experience=details.get("base_experience") or 0,
               stats=[
                   PokemonStat(
                       name=translate_stat(stat["stat"]["name"]),
                       value=stat["base_stat"],
                   )
                   for stat in details.get("stats", [])
               ],
               types=[
                   PokemonType(
                       key=entry["type"]["name"],
                       label=translate_type(entry["type"]["name"]),
                   )
                   for entry in details.get("types", [])
               ],
           )
       )


   collection.sort(key=lambda p: p.id)


   if not collection:
       raise RuntimeError("No se pudo cargar ningún Pokémon desde PokeAPI.")


   return collection

def sign_in_anonymously_with_firebase() -> dict:
   """
   Crea una sesión anónima en Firebase Auth usando la REST API.
   Devuelve, entre otros:
   - idToken
   - refreshToken
   - localId
   """
   url = (
       "AQUI_VA_TU_ENDPOINT_DE_AUTH_FIREBASE"
       f"?key={FIREBASE_API_KEY}"
   )
   payload = {"returnSecureToken": True}
   return post_json(url, payload)




def firestore_auth_headers(firebase_id_token: str) -> dict:
   return {
       "Authorization": f"Bearer {firebase_id_token}",  # Aqui va tu token de sesion
   }




def firestore_document_url(path: str) -> str:
   return (
       f"AQUI_VA_TU_ENDPOINT_DE_FIRESTORE/{FIREBASE_PROJECT_ID}"
       f"/databases/(default)/documents/{path}"
   )




def firestore_list_favorites(firebase_id_token: str, uid: str) -> set[int]:
   url = firestore_document_url(f"users/{uid}/favorites")
   try:
       payload = fetch_json(url, headers=firestore_auth_headers(firebase_id_token))
   except RuntimeError as exc:
       error_text = str(exc)
       if "HTTP 404" in error_text:
           return set()
       raise


   docs = payload.get("documents", [])
   result: set[int] = set()


   for doc in docs:
       fields = doc.get("fields", {})
       pokemon_id_raw = fields.get("pokemonId", {}).get("integerValue")
       if pokemon_id_raw is not None:
           try:
               result.add(int(pokemon_id_raw))
           except Exception:
               pass


   return result




def firestore_save_favorite(firebase_id_token: str, uid: str, pokemon: Pokemon) -> dict:
   url = firestore_document_url(f"users/{uid}/favorites/{pokemon.id}")
   payload = {
       "fields": {
           "pokemonId": {"integerValue": str(pokemon.id)},
           "name": {"stringValue": pokemon.name},
           "displayName": {"stringValue": pokemon.display_name},
           "image": {"stringValue": pokemon.image},
       }
   }
   return patch_json(url, payload, headers=firestore_auth_headers(firebase_id_token))




def firestore_delete_favorite(firebase_id_token: str, uid: str, pokemon_id: int) -> None:
   url = firestore_document_url(f"users/{uid}/favorites/{pokemon_id}")
   try:
       delete_request(url, headers=firestore_auth_headers(firebase_id_token))
   except RuntimeError as exc:
       if "HTTP 404" in str(exc):
           return
       raise

   def build_stat_rows(stats: list[PokemonStat]) -> ft.Column:
       rows: list[ft.Control] = []
       for stat in stats:

           rows.append(
               ft.Column(
                   spacing=6,
                   controls=[
                       ft.Row(
                           alignment=ft.MainAxisAlignment.SPACE_BETWEEN,
                           controls=[
                               ft.Text(stat.name, color="#132038", weight=ft.FontWeight.W_600),
                               ft.Text(str(stat.value), color="#132038", weight=ft.FontWeight.BOLD),
                           ],
                       ),
                       ft.ProgressBar(
                           value=percentage / 100,
                           color="#4F7EFF",
                           bgcolor="#DBE4F2",
                           bar_height=12,
                       ),
                   ],
               )
           )
       return ft.Column(spacing=12, controls=rows)

       page.update()


   def filter_pokemon(_: ft.ControlEvent | None = None) -> None:
       nonlocal filtered_collection, selected_pokemon_id


       query = (search_field.value or "").strip().lower()


       filtered_collection = [
           pokemon
           for pokemon in pokemon_collection
           if query in pokemon.name.lower()
           or query in pokemon.display_name.lower()
           or query in str(pokemon.id)
       ]


       if not any(pokemon.id == selected_pokemon_id for pokemon in filtered_collection):
           selected_pokemon_id = filtered_collection[0].id if filtered_collection else None


       render_pokemon_list()
       render_featured(get_selected_pokemon())
       update_counters()
       page.update()


   # ---------- FIREBASE FAVORITOS ----------
   async def load_favorites() -> None:
       nonlocal favorite_ids
       if not firebase_uid or not firebase_id_token:
           favorite_ids = set()
           return


       favorite_ids = await asyncio.to_thread(
           firestore_list_favorites,
           firebase_id_token,
           firebase_uid,
       )


   def toggle_favorite(pokemon: Pokemon) -> None:
       nonlocal favorite_status_message, favorite_ids


       if not firebase_uid or not firebase_id_token:
           show_snack("Firebase no está disponible para guardar favoritos.", "#8B1E2D")
           return


       async def persist_and_refresh() -> None:
           nonlocal favorite_status_message, favorite_ids
           try:
               if pokemon.id in favorite_ids:
                   await asyncio.to_thread(
                       firestore_delete_favorite,
                       firebase_id_token,
                       firebase_uid,
                       pokemon.id,
                   )
                   favorite_ids.remove(pokemon.id)
                   favorite_status_message = f"{pokemon.display_name} se eliminó de favoritos."
               else:
                   await asyncio.to_thread(
                       firestore_save_favorite,
                       firebase_id_token,
                       firebase_uid,
                       pokemon,
                   )
                   favorite_ids.add(pokemon.id)
                   favorite_status_message = f"{pokemon.display_name} se guardó en favoritos."


               render_favorites()
               render_featured(get_selected_pokemon())
               show_snack(favorite_status_message, "#2F8754")
               page.update()
           except Exception as exc:
               show_snack(f"No se pudo actualizar Firebase: {exc}", "#8B1E2D")


       page.run_task(persist_and_refresh)


   # ---------- CARGA DE POKÉMON ----------
   def load_pokemon_data(_: ft.ControlEvent | None = None) -> None:
       page.run_task(load_pokemon_data_async)


   async def load_pokemon_data_async() -> None:
       nonlocal pokemon_collection, filtered_collection, selected_pokemon_id
       nonlocal favorite_status_message, is_loading


       if is_loading:
           return


       is_loading = True
       refresh_button.disabled = True
       refresh_button.content = "Cargando..."
       status_banner.value = "Cargando Pokémon desde PokeAPI..."
       featured_container.content = ft.Container(
           alignment=ft.Alignment(0, 0),
           padding=30,
           content=ft.ProgressRing(color="#4EC7FF"),
       )
       page.update()


       try:
           pokemon_collection = await build_pokemon_collection_async()
           filtered_collection = list(pokemon_collection)
           selected_pokemon_id = pokemon_collection[0].id if pokemon_collection else None
           favorite_status_message = ""


           await load_favorites()
           render_pokemon_list()
           render_favorites()
           render_featured(get_selected_pokemon())
           update_counters()
           status_banner.value = f"Datos cargados: {len(pokemon_collection)} Pokémon."
       except Exception as exc:
           status_banner.value = f"Error al cargar: {exc}"
           featured_container.content = ft.Container(
               bgcolor="#FFFFFF",
               border_radius=18,
               padding=18,
               content=ft.Text(str(exc), color="#132038"),
           )
           pokemon_list_column.controls.clear()
           favorites_column.controls.clear()
           update_counters()
       finally:
           is_loading = False
           refresh_button.disabled = False
           refresh_button.content = "Recargar"
           page.update()


   async def initialize_firebase_session() -> None:
       nonlocal firebase_id_token, firebase_refresh_token, firebase_uid


       if "AQUI_VA_TU" in FIREBASE_API_KEY:
           raise RuntimeError("Falta configurar FIREBASE_API_KEY.")


       session_data = await asyncio.to_thread(sign_in_anonymously_with_firebase)
       firebase_id_token = session_data.get("idToken")
       firebase_refresh_token = session_data.get("refreshToken")
       firebase_uid = session_data.get("localId")


       if not firebase_id_token or not firebase_uid:
           raise RuntimeError("Firebase no devolvió una sesión anónima válida.")
