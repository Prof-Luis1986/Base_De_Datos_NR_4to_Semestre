🟦 Práctica 5

Cacería de Datos: Sistema visual con Pygame y MongoDB Atlas

🎯 Propósito

Desarrollar una aplicación funcional en Python con Pygame, basada en un código base obligatorio, que permita capturar, almacenar y gestionar registros con imágenes convertidas a Base64 en MongoDB Atlas, aplicando el método T.I.R.E. y realizando modificaciones funcionales reales.

⚠️ Condiciones obligatorias

  • Debes usar el código base proporcionado.
  • Debes mantener su funcionamiento.
  • No se aceptan proyectos desde cero.
  • No se aceptan proyectos sin modificaciones.
  • Debes modificar el sistema funcionalmente.
  • Debes aplicar el método T.I.R.E.

🔧 Paso 2 – Código base obligatorio

Este código es el punto de partida. Debe ser modificado y debe seguir funcionando. 

import pygame
import sys
import base64
from pymongo import MongoClient
from datetime import datetime

# =========================
# MONGODB
# =========================
MONGO_URI = "TU_URI_AQUI"
client = MongoClient(MONGO_URI)

db = client["caceria_datos"]
coleccion = db["registros"]

# =========================
# FUNCIONES
# =========================
def imagen_a_base64(ruta):
    try:
        with open(ruta, "rb") as archivo:
            return base64.b64encode(archivo.read()).decode("utf-8")
    except:
        return ""

def guardar_registro(data):
    coleccion.insert_one(data)

def obtener_registros():
    return list(coleccion.find({}, {"foto_base64": 0}))

def actualizar_registro(nombre, nuevos_datos):
    coleccion.update_one(
        {"nombre_lugar": nombre},
        {"$set": nuevos_datos}
    )

def eliminar_registro(nombre):
    coleccion.delete_one({"nombre_lugar": nombre})

# =========================
# PYGAME
# =========================
pygame.init()

ANCHO, ALTO = 1000, 650
pantalla = pygame.display.set_mode((ANCHO, ALTO))
pygame.display.set_caption("Cacería de Datos")

fuente = pygame.font.SysFont("Arial", 22)
reloj = pygame.time.Clock()

BLANCO = (255,255,255)
NEGRO = (30,30,30)
AZUL = (70,120,255)
GRIS = (230,230,230)
VERDE = (50,170,90)
ROJO = (200,70,70)

# =========================
# INPUT BOX
# =========================
class InputBox:
    def __init__(self, x, y, w, h):
        self.rect = pygame.Rect(x, y, w, h)
        self.texto = ""
        self.activo = False

    def manejar_evento(self, evento):
        if evento.type == pygame.MOUSEBUTTONDOWN:
            self.activo = self.rect.collidepoint(evento.pos)

        if evento.type == pygame.KEYDOWN and self.activo:
            if evento.key == pygame.K_BACKSPACE:
                self.texto = self.texto[:-1]
            else:
                self.texto += evento.unicode

    def dibujar(self, pantalla):
        pygame.draw.rect(pantalla, BLANCO, self.rect)
        pygame.draw.rect(pantalla, AZUL if self.activo else NEGRO, self.rect, 2)
        texto_render = fuente.render(self.texto, True, NEGRO)
        pantalla.blit(texto_render, (self.rect.x+5, self.rect.y+8))

# =========================
# CAMPOS
# =========================
campos = {
    "nombre_lugar": InputBox(300, 80, 500, 35),
    "ubicacion": InputBox(300, 130, 500, 35),
    "tipo_lugar": InputBox(300, 180, 500, 35),
    "nivel_actividad": InputBox(300, 230, 500, 35),
    "comentario": InputBox(300, 280, 500, 35),
    "ruta_imagen": InputBox(300, 330, 500, 35),
}

mensaje = ""
registros = []

# =========================
# BOTONES
# =========================
btn_guardar = pygame.Rect(200, 400, 150, 50)
btn_consultar = pygame.Rect(380, 400, 150, 50)
btn_actualizar = pygame.Rect(560, 400, 150, 50)
btn_eliminar = pygame.Rect(740, 400, 150, 50)

# =========================
# LOOP PRINCIPAL
# =========================
while True:
    pantalla.fill(GRIS)

    for evento in pygame.event.get():
        if evento.type == pygame.QUIT:
            pygame.quit()
            sys.exit()

        for campo in campos.values():
            campo.manejar_evento(evento)

        if evento.type == pygame.MOUSEBUTTONDOWN:

            # GUARDAR
            if btn_guardar.collidepoint(evento.pos):
                try:
                    ruta = campos["ruta_imagen"].texto
                    documento = {
                        "nombre_lugar": campos["nombre_lugar"].texto,
                        "ubicacion": campos["ubicacion"].texto,
                        "tipo_lugar": campos["tipo_lugar"].texto,
                        "nivel_actividad": int(campos["nivel_actividad"].texto),
                        "comentario": campos["comentario"].texto,
                        "foto_base64": imagen_a_base64(ruta),
                        "fecha": datetime.utcnow()
                    }
                    guardar_registro(documento)
                    mensaje = "Registro guardado"
                except:
                    mensaje = "Error al guardar"

            # CONSULTAR
            if btn_consultar.collidepoint(evento.pos):
                registros = obtener_registros()
                mensaje = f"{len(registros)} registros encontrados"

            # ACTUALIZAR
            if btn_actualizar.collidepoint(evento.pos):
                actualizar_registro(
                    campos["nombre_lugar"].texto,
                    {"comentario": campos["comentario"].texto}
                )
                mensaje = "Registro actualizado"

            # ELIMINAR
            if btn_eliminar.collidepoint(evento.pos):
                eliminar_registro(campos["nombre_lugar"].texto)
                mensaje = "Registro eliminado"

    # DIBUJAR CAMPOS
    for campo in campos.values():
        campo.dibujar(pantalla)

    # BOTONES
    pygame.draw.rect(pantalla, VERDE, btn_guardar)
    pygame.draw.rect(pantalla, AZUL, btn_consultar)
    pygame.draw.rect(pantalla, (240,180,60), btn_actualizar)
    pygame.draw.rect(pantalla, ROJO, btn_eliminar)

    pantalla.blit(fuente.render("Guardar", True, BLANCO), (230,415))
    pantalla.blit(fuente.render("Consultar", True, BLANCO), (400,415))
    pantalla.blit(fuente.render("Actualizar", True, BLANCO), (570,415))
    pantalla.blit(fuente.render("Eliminar", True, BLANCO), (760,415))

    pantalla.blit(fuente.render(mensaje, True, NEGRO), (50, 500))

    pygame.display.flip()
    reloj.tick(60)
